set port /dev/ttyUSB0

set line /dev/ttyUSB0
set speed 115200
set carrier-watch off
set handshake none
set flow-control none
set serial 8n1
robust
set file type bin
set file name lit
set rec pack 1000
set send pack 1000
set window 5

connect

set retry 0
