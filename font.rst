list enabled fonts ::

   fc-list | grep -ie noto | grep -ie jp

list enabled and disabled configs ::

   fc-conflist \
      | sed \
         -e 's#/usr/share/fontconfig/conf.avail#//usr.avail#g' \
         -e 's#/etc/fonts/conf.d#//etc.d#g' \
      | sort -r \
      | column -s: -t | less -SRM +%

refresh cache ::

   sudo fc-cache -f -r -v; fc-cache -f -r -v
